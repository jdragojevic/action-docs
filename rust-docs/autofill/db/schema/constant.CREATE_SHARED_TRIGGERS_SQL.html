const CREATE_SHARED_TRIGGERS_SQL: &str = "\n-- This Source Code Form is subject to the terms of the Mozilla Public\n-- License, v. 2.0. If a copy of the MPL was not distributed with this\n-- file, You can obtain one at http://mozilla.org/MPL/2.0/.\n\n-- This file defines triggers shared between the main and Sync connections.\n\nCREATE TEMP TRIGGER addresses_data_afterinsert_trigger\nAFTER INSERT ON addresses_data\nFOR EACH ROW WHEN NEW.guid IN (SELECT guid FROM addresses_tombstones)\nBEGIN\n    SELECT RAISE(FAIL, \'guid exists in `addresses_tombstones`\');\nEND;\n\nCREATE TEMP TRIGGER addresses_tombstones_afterinsert_trigger\nAFTER INSERT ON addresses_tombstones\nWHEN NEW.guid IN (SELECT guid FROM addresses_data)\nBEGIN\n    SELECT RAISE(FAIL, \'guid exists in `addresses_data`\');\nEND;\n\nCREATE TEMP TRIGGER addresses_tombstones_create_trigger\nAFTER DELETE ON addresses_data\nWHEN OLD.guid IN (SELECT guid FROM addresses_mirror)\nBEGIN\n    INSERT INTO addresses_tombstones(guid, time_deleted)\n    VALUES (OLD.guid, now());\nEND;\n\nCREATE TEMP TRIGGER credit_cards_data_afterinsert_trigger\nAFTER INSERT ON credit_cards_data\nFOR EACH ROW WHEN NEW.guid IN (SELECT guid FROM credit_cards_tombstones)\nBEGIN\n    SELECT RAISE(FAIL, \'guid exists in `credit_cards_tombstones`\');\nEND;\n\nCREATE TEMP TRIGGER credit_cards_tombstones_afterinsert_trigger\nAFTER INSERT ON credit_cards_tombstones\nWHEN NEW.guid IN (SELECT guid FROM credit_cards_data)\nBEGIN\n    SELECT RAISE(FAIL, \'guid exists in `credit_cards_data`\');\nEND;\n\nCREATE TEMP TRIGGER credit_cards_tombstones_create_trigger\nAFTER DELETE ON credit_cards_data\nWHEN OLD.guid IN (SELECT guid FROM credit_cards_mirror)\nBEGIN\n    INSERT INTO credit_cards_tombstones(guid, time_deleted)\n    VALUES (OLD.guid, now());\nEND;\n";
